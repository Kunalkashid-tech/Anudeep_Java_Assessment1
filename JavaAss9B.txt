package Anudeep_pro.java;
   // Java program to find odd numbers between 1 to 10 numbers using while loop. 
public class Q9_B_Java_Ass {

    public static void main(String[] args) {
		
         System.out.println("Odd Numbers");

	int i = 0;
		
	while(i <= 10) {
			
	if(i % 2 != 0) {
				
        System.out.println(i);
			}
	i++;
		}
	}
}
